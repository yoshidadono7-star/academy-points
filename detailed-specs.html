ComboDB.buildContext() — 判定用コンテキストの組み立て
// 学習セッション完了時に呼ばれる
async buildContext(studentId, session) {
  const today = getToday();
  const todayLogs = await StudyLogsDB.getByDate(today);
  const myTodayLogs = todayLogs.filter(l => l.studentId === studentId);
  const student = await StudentsDB.getById(studentId);
  const todayCondition = await ConditionsDB.getToday();
  const todayTasks = await TasksDB.getCompletions(studentId);
  const activeBoss = await RaidBossDB.getActive();
  
  // 今週の出席日数を計算
  const weekStart = new Date();
  weekStart.setDate(weekStart.getDate() - weekStart.getDay());
  const weekLogs = await StudyLogsDB.getByDateRange(
    weekStart.toISOString().split('T')[0], today
  );
  const weekDays = new Set(
    weekLogs.filter(l => l.studentId === studentId).map(l => l.date)
  ).size;

  return {
    // セッション情報
    subject: session.subject,
    duration: session.duration,
    location: session.location,
    method: session.method,
    pomodoroCount: session.pomodoroCount || 0,
    sessionStartHour: new Date(session.startedAt).getHours(),

    // 本日の集計
    todaySubjects: [...new Set(myTodayLogs.map(l => l.subject))],
    todayPomodoros: myTodayLogs.reduce((s,l) => s + (l.pomodoroCount||0), 0),
    studiedToday: myTodayLogs.length > 0,
    conditionToday: todayCondition.some(c => c.studentId === studentId),
    taskCompletedToday: todayTasks.some(t => t.date === today),

    // ストリーク
    currentStreak: student.streak || 0,
    missionStreak: student.missionStreak || 0,

    // 週間
    weeklyAttendance: weekDays,

    // 苦手科目（生徒プロフィールに設定）
    weakSubjectMinutes: myTodayLogs
      .filter(l => l.subject === student.weakSubject)
      .reduce((s,l) => s + (l.duration||0), 0),

    // ボス連動
    isWeaknessSubject: activeBoss && session.subject === activeBoss.weakness,
  };
}
3. 判定フロー
学習セッション完了 ↓ ComboDB.buildContext(studentId, session) ↓ COMBO_DEFINITIONS を全件ループ ↓ 各コンボに対して: ├─ combo.condition(ctx) === true ? │ ├─ NO → スキップ │ └─ YES → クールダウンチェック │ ├─ cooldown='daily' → 今日すでに発動済み? → スキップ │ ├─ cooldown='weekly' → 今週すでに発動済み? → スキップ │ └─ 発動OK! │ ├─ comboDiscoveriesに記録 (初回なら新規, 既存ならtriggerCount++) │ ├─ comboTriggersにログ追加 │ ├─ reward.type === 'bonus' → ポイント加算 │ ├─ reward.type === 'multiplier' → セッションポイントに倍率適用 │ ├─ reward.type === 'boss_multiplier' → ボスダメージに倍率適用 │ └─ 発動結果を配列に追加 ↓ 発動したコンボ一覧を返す → student.htmlの完了画面で表示 ↓ 初回発見のコンボがあれば → 特別演出（コンフェッティ + 図鑑登録）

主要メソッド
const ComboDB = {
  // コンテキスト構築
  async buildContext(studentId, session) { ... },

  // 全コンボ判定（セッション完了時に呼ぶ）
  async checkAll(studentId, session) {
    const ctx = await this.buildContext(studentId, session);
    const triggered = [];
    for (const combo of COMBO_DEFINITIONS) {
      if (!combo.condition(ctx)) continue;
      if (await this.isOnCooldown(studentId, combo)) continue;
      const isNew = await this.record(studentId, combo.id);
      await this.logTrigger(studentId, combo, session);
      triggered.push({ ...combo, isFirstDiscovery: isNew });
    }
    return triggered; // [{id, name, icon, reward, isFirstDiscovery}, ...]
  },

  // クールダウン判定
  async isOnCooldown(studentId, combo) { ... },

  // 発見記録（初回ならtrue, 既存ならfalse）
  async record(studentId, comboId) { ... },

  // 発動ログ
  async logTrigger(studentId, combo, session) { ... },

  // 図鑑取得（student.html用）
  async getDiscoveries(studentId) { ... },

  // 発見率（ストーリー用: 全生徒の平均発見数）
  async getClassDiscoveryRate() { ... },
};

BadgesDBとの設計パターンの違い: バッジは「一度獲得したら永久」。コンボは「毎回発動するがクールダウンあり」。 バッジはcheckAndAward()で条件チェック+付与が一体。コンボはcheckAll()で判定→返却値を元にUI側で報酬適用する分離型。 これはコンボの報酬タイプ（bonus/multiplier/boss_multiplier）が複数あり、適用先が違うため。

2. 抽選アルゴリズム
DailyMissionsDB.draw()
async draw(studentId) {
  const today = getToday();
  
  // 今日の抽選済みチェック
  const existing = await db.collection('dailyMissionDraws')
    .where('studentId', '==', studentId)
    .where('date', '==', today).limit(1).get();
  if (!existing.empty) return existing.docs[0].data(); // 既存を返す
  
  // レアリティ抽選（3枠独立）
  const drawn = [];
  for (let i = 0; i < 3; i++) {
    const rarity = this.rollRarity();
    const pool = DAILY_MISSION_POOL.filter(
      m => m.rarity === rarity && !drawn.some(d => d.missionId === m.id)
    );
    if (pool.length === 0) continue; // fallback: common
    const pick = pool[Math.floor(Math.random() * pool.length)];
    drawn.push({ missionId: pick.id, rarity: pick.rarity });
  }
  
  const doc = {
    studentId, date: today, missions: drawn,
    accepted: null, completed: false, completedAt: null,
    createdAt: firebase.firestore.FieldValue.serverTimestamp()
  };
  await db.collection('dailyMissionDraws').add(doc);
  return doc;
},

rollRarity() {
  const r = Math.random() * 100;
  if (r < 1)  return 'legendary';  // 1%
  if (r < 5)  return 'epic';       // 4%
  if (r < 20) return 'rare';       // 15%
  if (r < 50) return 'uncommon';   // 30%
  return 'common';                  // 50%
}
3. 達成判定フロー
生徒がログイン → draw(studentId) → 3枚のカード表示 ↓ 生徒が1枚を選択 → accept(drawId, missionId) ↓ 学習セッション完了時 or 1日の最終チェック時: buildDayStats(studentId) でその日の全データを集計 ↓ 受注中ミッションの condition(dayStats) を評価 ↓ true → complete(drawId) → ポイント付与 → student.missionStreak++ false → 翌日に持ち越さない（日単位で完結）
既存のMissionsDB（月次ミッション）との共存: 月次ミッション = クラス全体の大きな目標（既存のまま）。 デイリーミッション = 個人の日替わり小目標（今回新規追加）。 別コレクション・別モジュールで完全に独立。両方同時に進行する。

function getStation(totalPoints) {
  for (let i = SPACE_STATIONS.length - 1; i >= 0; i--) {
    if (totalPoints >= SPACE_STATIONS[i].points) return SPACE_STATIONS[i];
  }
  return SPACE_STATIONS[0];
}

function getStationProgress(totalPoints) {
  const current = getStation(totalPoints);
  const currentIdx = SPACE_STATIONS.indexOf(current);
  if (currentIdx >= SPACE_STATIONS.length - 1) return { percent: 100, remaining: 0 };
  const next = SPACE_STATIONS[currentIdx + 1];
  const range = next.points - current.points;
  const progress = totalPoints - current.points;
  return {
    current, next,
    percent: Math.floor(progress / range * 100),
    remaining: next.points - totalPoints
  };
}

2. ダメージ計算式（改修）
calculateBossDamage() — 新しいダメージ計算
function calculateBossDamage(session, boss, comboResults) {
  let damage = session.duration; // ベース = 学習分数

  // 1. 属性倍率
  if (session.subject === boss.weakness) {
    damage *= 2.0;  // 弱点ヒット
  } else if (session.subject === boss.resistance) {
    damage *= 0.5;  // 耐性で半減
  }
  // それ以外は×1.0

  // 2. コンボ倍率（boss_multiplier型のコンボが発動していれば）
  const bossCombo = comboResults.find(c => c.reward.type === 'boss_multiplier');
  if (bossCombo) {
    damage *= bossCombo.reward.value;
  }

  // 3. ポモドーロボーナス
  damage += (session.pomodoroCount || 0) * 10;

  // 4. 特殊能力による補正
  if (boss.specialAbility) {
    switch (boss.specialAbility.type) {
      case 'min_duration':
        if (session.duration < boss.specialAbility.value) return 0;
        break;
      case 'enrage':
        const hpPercent = (boss.currentHp / boss.maxHp) * 100;
        if (hpPercent <= boss.specialAbility.value) damage *= 0.5;
        break;
      case 'night_boost':
        const hour = new Date().getHours();
        if (hour >= 20) damage *= boss.specialAbility.value;
        break;
    }
  }

  return Math.floor(damage);
}

// friday_regen: 金曜に実行
async function processBossSpecials() {
  const bosses = await RaidBossDB.getActive();
  const today = new Date();
  const dayOfWeek = today.getDay(); // 0=日, 5=金

  for (const boss of bosses) {
    if (!boss.specialAbility) continue;
    switch (boss.specialAbility.type) {
      case 'friday_regen':
        if (dayOfWeek === 5) {
          const newHp = Math.min(boss.maxHp, 
            boss.currentHp + boss.specialAbility.value);
          await db.collection('raidBosses').doc(boss.id)
            .update({ currentHp: newHp });
        }
        break;
      case 'weekly_rotate':
        if (dayOfWeek === 1) { // 月曜
          await db.collection('raidBosses').doc(boss.id).update({
            weakness: boss.resistance,
            resistance: boss.weakness
          });
        }
        break;
    }
  }
}

2. ルート進捗の更新ロジック
StoryDB.updateProgress()
const StoryDB = {
  // アクティブチャプター取得
  async getActive() {
    const snap = await db.collection('storyChapters')
      .where('status', '==', 'active').limit(1).get();
    return snap.empty ? null : { id: snap.docs[0].id, ...snap.docs[0].data() };
  },

  // 進捗更新（学習セッション完了時、日次バッチなど）
  async updateProgress() {
    const chapter = await this.getActive();
    if (!chapter) return;

    for (const route of chapter.routes) {
      if (route.metric === 'fallback') continue;

      let currentValue = 0;
      switch (route.metric) {
        case 'total_pomodoros':
          // 今月の全生徒ポモドーロ合計
          const monthStart = chapter.month + '-01';
          const logs = await StudyLogsDB.getByDateRange(monthStart, getToday());
          currentValue = logs.reduce((s, l) => s + (l.pomodoroCount || 0), 0);
          break;

        case 'combo_discovery_rate':
          // 全生徒のうちコンボ3種以上発見した割合(%)
          const students = await StudentsDB.getAll();
          let qualified = 0;
          for (const st of students) {
            const discoveries = await ComboDB.getDiscoveries(st.id);
            if (discoveries.length >= 3) qualified++;
          }
          currentValue = students.length > 0
            ? Math.floor(qualified / students.length * 100) : 0;
          break;

        case 'total_study_hours':
          // 今月の全生徒合計学習時間
          const mLogs = await StudyLogsDB.getByDateRange(
            chapter.month + '-01', getToday());
          currentValue = Math.floor(
            mLogs.reduce((s, l) => s + (l.duration || 0), 0) / 60);
          break;

        case 'boss_defeated':
          // 今月のボスが撃破されたか (0 or 1)
          const bosses = await RaidBossDB.getAll();
          const monthBoss = bosses.find(
            b => b.month === chapter.month && b.status === 'defeated');
          currentValue = monthBoss ? 1 : 0;
          break;
      }

      route.currentValue = currentValue;
    }

    // Firestore更新
    await db.collection('storyChapters').doc(chapter.id)
      .update({ routes: chapter.routes });

    return chapter;
  },

  // 月末の結末判定
  async resolveChapter() {
    const chapter = await this.getActive();
    if (!chapter) return;

    // ルートAから順に判定（最初に達成したルートが採用）
    let achievedRoute = null;
    for (const route of chapter.routes) {
      if (route.metric === 'fallback') continue;
      const percent = route.targetValue > 0
        ? (route.currentValue / route.targetValue) * 100 : 0;
      if (percent >= 100) {
        achievedRoute = route.id;
        break;
      }
    }
    // どれも未達成ならfallback
    if (!achievedRoute) {
      achievedRoute = chapter.routes.find(r => r.metric === 'fallback')?.id;
    }

    await db.collection('storyChapters').doc(chapter.id).update({
      status: 'completed',
      achievedRoute,
      completedAt: firebase.firestore.FieldValue.serverTimestamp()
    });

    // 報酬配布
    const route = chapter.routes.find(r => r.id === achievedRoute);
    if (route?.reward?.points) {
      const students = await StudentsDB.getAll();
      for (const st of students) {
        await StudentsDB.addPoints(st.id, route.reward.points);
      }
    }

    return { chapter, achievedRoute, route };
  },

  // チャプター作成（先生がindex.htmlから）
  async create(data) { ... },
};
